mechanism_status: mixed
mechanism_status_notes: 'Documented mechanism is winner-takes-all over EDA-evaluated Verilog submissions,
  but the live validator defaults to burning the bulk of emissions: only --miner_emission_percentage (CLI
  default 10%) goes to the current winner and the remainder is sent to UID 0, and emissions are fully
  burned whenever there is no active challenge, no qualifying submission beating the baseline, an emissions
  ban from the challenge server, or any error/timeout.'
subnet:
  name: ChipForge
  owner_repo: TatsuProject__ChipForge_SN84
  netuid: 84
  implementation_lang: python
task:
  summary: Miners download an on-chain hardware-design challenge spec and submit complete Verilog/SystemVerilog
    implementations (packaged as a ZIP). Validators download submissions in batches, evaluate them with
    an external EDA server (Verilator functionality, Openlane synthesis/place-and-route), and reward the
    single highest-scoring submission that passes functional and overall gates and beats the challenge
    baseline. Winner-takes-all.
  submission_format:
  - source_code
scoring_signals:
- name: overall_score
  metric_kind: composite
  direction: higher_is_better
  weight: null
  normalization: none
  notes: The EDA server's final_score.overall is used as the per-submission score for ranking and winner
    selection. README states Functionality is the only currently active component (100% weight); area/delay/power
    are documented as TBD/future.
- name: functional_gate
  metric_kind: accuracy
  direction: higher_is_better
  weight: null
  normalization: none
  notes: Boolean gate from final_score.functional_gate. Must be True or the submission is dropped from
    weight calculation and winner comparison.
- name: overall_gate
  metric_kind: accuracy
  direction: higher_is_better
  weight: null
  normalization: none
  notes: Boolean gate from final_score.overall_gate. Must be True or the submission is dropped.
ground_truth_sources:
- kind: deterministic_dataset
  description: Challenge test cases / testbenches downloaded from the ChipForge Challenge Server, plus
    a server-provided winner_baseline_score that a submission must beat to qualify for reward.
  source_identifier: ChipForge Challenge Server API (/api/v1/challenges)
  trust_model: trusted
  notes: Functionality verified via Verilator testbench pass/fail; synthesis metrics via Openlane. Baseline
    score and ban_emissions flag are controlled by the challenge server.
aggregation:
  method: winner_take_all
  composition: gated
  normalization: none
  temperature: null
  decay_rate: null
  min_weight_floor: null
  burn_allocation:
    enabled: true
    address_or_uid: '0'
    fraction: null
    dynamic: true
    notes: When a winner is rewarded via set_winner_weights, miner gets miner_emission_percentage/100
      and burn UID 0 gets the remainder (1 - miner_fraction). In all burn paths set_burn_weights gives
      UID 0 weight 1.0 and every other UID 0.0.
  notes: Per batch, gated submissions are sorted by overall_score and the top one gets weight 1.0, others
    0.0 (calculate_weights_from_hotkeys). At the validator level rewards are routed through the EmissionManager
    phase machine and the miner/burn split.
weight_setting:
  cadence: continuous
  method: Validator loop runs every 10s. set_weights_with_ban_check chooses between set_burn_weights (burn
    to UID 0) and set_winner_weights (split between winner UID and UID 0 by miner_emission_percentage).
    A new champion that beats the baseline snapshot is set to weight 1.0 directly. Burn occurs on ban,
    no challenge, no qualifying winner, or errors.
  on_chain_call: set_weights
  tempo_or_interval: 10s loop
  smoothing:
    kind: none
    alpha: null
    window: null
    notes: No EMA/SMA smoothing of final weights; weights are set directly each cycle. A winner retains
      reward for TOTAL_HOURS_FOR_WINNER_REWARD (env/server, local default 0.5h) after challenge end via
      the EmissionManager reward timer.
anti_gaming:
- kind: deterministic_check
  enforcement: rejection
  description: functional_gate and overall_gate (both must be True) gate submissions out of weight calculation
    and winner comparison; submissions are also re-checked against the winner_baseline_score snapshot
    before being crowned champion.
- kind: stake_weighting
  enforcement: barrier
  description: Banned coldkeys list synced from the challenge server; a banned winner's coldkey causes
    all emissions to be burned (set_burn_weights) instead of rewarding that UID.
- kind: rate_limit
  enforcement: rejection
  description: 'README: challenge server accepts 5 requests per IP per hour; one hotkey may submit a maximum
    of 5 solutions per challenge; max ZIP size 10MB.'
sub_competitions:
  structure: single
  stages: []
  count: null
  notes: One active challenge at a time (rotated periodically); challenge categories include RISC-V processors
    and AI accelerators, but only one challenge is scored per period.
