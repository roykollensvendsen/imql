subnet:
  name: NeuralAI (3D Asset Generator)
  owner_repo: goneuralai__neural-subnet
  netuid: 46
  implementation_lang: python
task:
  summary: A miner receives a text prompt and generates a 3D mesh object, returning it (plus a preview
    image) to the validator. The validator renders the mesh from multiple axes and scores it on how well
    the rendered images and preview match the prompt (semantic similarity), how well the rendered views
    match the preview (CLIP image similarity), and overall image quality. Generation quality is rewarded;
    generation speed can be optionally weighted in.
  submission_format:
  - content_link
  - api_response
scoring_signals:
- name: preview_prompt_semantic_similarity
  metric_kind: similarity
  direction: higher_is_better
  weight: 0.25
  normalization: none
  notes: 'S0 = semantic similarity between prompt vector and preview-image description vector (only computed
    if quality Q0 > 0.15). Contributes 0.25 to total_score. Acts as a gate too: if S0 < 0.23 the score
    is 0.'
- name: render_prompt_semantic_similarity
  metric_kind: similarity
  direction: higher_is_better
  weight: 0.5
  normalization: none
  notes: S_geo = geometric mean of outlier-filtered semantic similarities between the prompt and the rendered-image
    descriptions. Contributes 0.5 to total_score.
- name: render_preview_clip_similarity
  metric_kind: similarity
  direction: higher_is_better
  weight: 0.3
  normalization: none
  notes: R_geo = geometric mean of outlier-filtered CLIP similarities between the preview image and each
    rendered image. Contributes 0.3 to total_score.
- name: image_quality
  metric_kind: similarity
  direction: higher_is_better
  weight: 0.1
  normalization: none
  notes: Q_geo = geometric mean of outlier-filtered per-image quality scores of the pre-render ('before')
    images. Contributes 0.1 to total_score.
- name: generation_time
  metric_kind: latency
  direction: lower_is_better
  weight: null
  normalization: min_max
  notes: Process time per miner is normalized (x/max)^2 across miners; final score blends validation score
    with (1 - time_score) using validator.time_rate. Default time_rate is 0, so by default speed has no
    effect.

    and an image-vision model, and gates on whether the output looks like a real object. Image-to-text
    descriptions are produced via Anthropic Claude 3.5 Haiku.
  source_identifier: validation/validation_endpoint.py (http endpoint, default http://127.0.0.1:8094)
  trust_model: oracle
  notes: 'Uses CLIP (image), text-similarity, quality, and image-vision (Claude) models. README: ''Claude
    API key is utilized for generating image-to-text outputs through Anthropic Claude 3.5 Haiku service.'''

  notes: 'Per-response total_score = S0*0.25 + S_geo*0.5 + R_geo*0.3 + Q_geo*0.1 (note weights sum to
    1.15, not 1). Hard gates zero the score: image not a real object => 0; S0<0.23 => 0; total_score<0.35
    => 0. Across miners, validation scores are normalized by (x/max)^2 (normalize()). final = s*(1-time_rate)
    + time_rate*(1-t) when a time score exists. EMA-updated base_scores are then transformed in set_weights
    as score**8 if score>0.4 else 0, and L1-normalized (raw_weights = scores / ||scores||_1).'
weight_setting:
  cadence: per_epoch
  method: base_scores (EMA of per-step rewards) are gated and steepened (score**8 if >0.4 else 0), then
    L1-normalized, processed via process_weights_for_netuid, converted to uint16 and emitted with set_weights.
  on_chain_call: set_weights
  tempo_or_interval: 100 blocks (neuron.epoch_length default 100, ~12s blocks)
  smoothing:
    kind: ema
    alpha: 0.1
    window: null
    notes: update_scores applies base_scores[i] = alpha*(0 if reward<0.1 else reward) + (1-alpha)*base_scores[i],
      with alpha = neuron.moving_average_alpha (default 0.1). base_scores below 4e-2 are zeroed.
anti_gaming:
- kind: deterministic_check
  enforcement: rejection
  description: 'Real-object gate: an image-vision model checks whether the rendered/preview images look
    like a real 3D object; if not, score returns 0.'
- kind: deterministic_check
  enforcement: rejection
  description: 'Threshold gates: S0 < 0.23 returns 0; total_score < 0.35 returns 0; rewards below 0.1
    are treated as 0 in the EMA update; base_scores below 4e-2 zeroed.'
- kind: deterministic_check
  enforcement: penalty
  description: Outlier filtering (IQR-based, threshold 1.1) on the per-image similarity/quality lists
    before taking geometric means, removing anomalous scores.
- kind: liveness_check
  enforcement: penalty
  description: Non-serving / vpermit-failing UIDs are given zero scores in update_scores (check_serving_axon
    + vpermit_tao_limit default 85000).
- kind: stake_weighting
  enforcement: barrier
  description: vpermit_tao_limit gating on which axons count as serving.
