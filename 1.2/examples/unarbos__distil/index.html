subnet:
  name: Distil — SN97 (GLM-5 / Kimi-K2.6 distillation)
  owner_repo: unarbos__distil
  netuid: 97
  implementation_lang: python
task:
  summary: Miners distill the teacher model (moonshotai/Kimi-K2.6, ~1T total / ~32B active MoE) into a
    smaller student (<=33B total params, Kimi-family DeepseekV3 architecture, vocab 163,840), upload it
    to HuggingFace, and commit the repo link on-chain (one permanent commitment per hotkey). Validators
    score every committed model on a multi-axis composite (composite.final = alpha*worst_K_mean + (1-alpha)*weighted)
    anchored on procedurally-generated v31 axes plus distillation/judge/discipline tiers, crown the highest
    composite.final as "king", and set on-chain weights to that king (winner-take-all, optionally split
    across recent kings).
  submission_format:
  - model_weights
  - commitment_hash
scoring_signals:
- name: composite_final
  metric_kind: composite
  direction: higher_is_better
  weight: 1.0
  normalization: none
  notes: Ranking key. composite.final = COMPOSITE_FINAL_BOTTOM_WEIGHT*worst_K_mean + (1-COMPOSITE_FINAL_BOTTOM_WEIGHT)*weighted.
    Code default alpha=0.85, K (WORST_3_MEAN_K) default 3; README states live values alpha=0.85, K=5 (env-overridden).
- name: worst_K_mean
  metric_kind: composite
  direction: higher_is_better
  weight: null
  normalization: none
  notes: Mean of the bottom-K non-broken per-axis values (K default 3). Anti-Goodhart pressure; dominates
    final via alpha=0.85. Surfaced as worst_3_mean in API.
- name: weighted
  metric_kind: composite
  direction: higher_is_better
  weight: null
  normalization: none
  notes: Convex combination Sum(w_i*axis_i)/Sum(w_i) over AXIS_WEIGHTS and tier weights. Each axis normalized
    to [0,1].
- name: kl_divergence
  metric_kind: loss
  direction: lower_is_better
  weight: null
  normalization: none
  notes: One distillation axis (top-20 sparse KL vs teacher); KL > maxKlThreshold (2.0) disqualifies.
    KL alone never crowns; reward=exp(-kl) only in the legacy distillation/validator/reward.py path.
- name: on_policy_rkl
  metric_kind: loss
  direction: lower_is_better
  weight: null
  normalization: none
  notes: Distillation tier axis, AXIS_WEIGHTS weight 0.30 (highest single axis weight).
- name: judge_probe
  metric_kind: llm_judgment
  direction: higher_is_better
  weight: null
  normalization: none
  notes: Quality tier; AXIS_WEIGHTS judge_probe 0.20 per README.

anti_gaming:
- kind: deduplication
  enforcement: rejection
  description: SHA256 hash of safetensors weights; identical weights to an existing model are blacklisted
    (COPY DQ) for that commitment. Earlier commitment by block number owns the hash.
- kind: plagiarism_detection
  enforcement: rejection
  description: 'Logit/activation fingerprinting: models with cosine similarity > 0.99999 (activationCopyThreshold)
    on per-position vectors over the first 2 prompts are flagged as functional copies even when hashes
    differ.'
- kind: data_augmentation
  enforcement: barrier
  description: Block-hash-seeded procedural v31 axes (no static answer key), Isomorphic Perturbation Testing
    (name rotation in v31_consistency_paraphrase), and GSM-NoOp topical-distractor injection (v31_math_robustness)
    defeat memorised canonical wordings.
- kind: deterministic_check
  enforcement: rejection
  description: 'Architecture/integrity precheck (no GPU): <=33B total params (MoE-aware counting), Kimi
    tokenizer vocab 163840, no quantization (GPTQ/AWQ/FP8 rejected), model must stay public and unchanged
    on HuggingFace (revision-pinned git-SHA integrity); failure = INVALID/REMOVED DQ.'
- kind: liveness_check
  enforcement: penalty
  description: Per-UID failure counts (failures.json); is_stale at max_failures default 3. thinking_collapse_probe
    / reasoning_density axis penalise models that loop without answering.
sub_competitions:
  structure: tournament
  stages:
  - name: precheck
    role: qualifier
    advancement_rule: No-GPU architecture/integrity/duplicate checks every epoch; models that fail are
      never sent to GPU.
  - name: per_uid_composite_eval
    role: main
    advancement_rule: Single-eval per commitment (FIFO by commit_block, capped at SINGLE_EVAL_MAX_PER_ROUND
      default 10) plus always-in reference baseline; compute composite.final.
  - name: king_dethronement
    role: boss
    advancement_rule: King = max composite.final; challenger dethrones only if challenger.final > incumbent.final*(1+SINGLE_EVAL_DETHRONE_MARGIN)
      (default 0.03; README 5%). King re-evaluated every round for paired fairness.
  count: null
  notes: King-of-the-hill architecture.
