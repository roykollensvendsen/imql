subnet:
  name: Compute Subnet (Lium.io)
  owner_repo: Datura-ai__compute-subnet
  netuid: 51
  implementation_lang: python
task:
  summary: Miners contribute GPU-equipped executor machines to a decentralized GPU-rental marketplace.
    Each cycle a validator SSHes into each executor, probes its hardware specs, runs a GraVal-style device-info-seeded
    matrix-multiplication challenge to prove the GPU is genuine, runs a VerifyX network/storage/memory
    probe, and checks rental status, price, collateral deposit and uptime. A per-executor mining score
    is the GPU-type rate weighted by the miner's share of that GPU type, gated by binary pass/fail checks
    and reduced by multiplicative penalties (sysbox, uptime). 91% of emission is burned; the remaining
    9% is distributed proportionally to mining score.
  submission_format:
  - compute_service
scoring_signals:
- name: gpu_type_rate
  metric_kind: hardware_spec
  direction: higher_is_better
  weight: null
  normalization: none
  notes: Per-GPU-model emission rate (gpu_portion) from GPU_MODEL_RATES; higher-tier data-center GPUs
    (H200, B200, etc.) carry larger rates, many consumer cards are 0.0.
- name: gpu_count_share
  metric_kind: hardware_spec
  direction: higher_is_better
  weight: null
  normalization: none
  notes: mining_score = score * gpu_portion * gpu_count / total_gpu_count — a miner's share of the total
    GPUs of that model online this cycle. Capped at MAX_GPU_COUNT=14.
- name: graval_matrix_challenge
  metric_kind: hardware_spec
  direction: higher_is_better
  weight: null
  normalization: none
  notes: 'Binary GPU-authenticity check: validator encrypts a device-info-seeded matrix challenge (DMCompVerify),
    executor must decrypt and return the embedded UUID. UUID mismatch => fail.'
- name: ema_verifyx_download_speed
  metric_kind: throughput
  direction: higher_is_better
  weight: null
  normalization: other
  notes: Exponential moving average (alpha=0.3) of VerifyX download speed in Mbps. Below MIN_VERIFYX_EMA_DOWNLOAD_SPEED_MBPS=100.0
    the executor fails; if EMA is missing for an unrented machine, actual_score=0.
- name: gpu_price_within_limit
  metric_kind: hardware_spec
  direction: target_band
  weight: null
  normalization: none
  notes: 'Binary gate: price_per_gpu must not exceed MACHINE_PRICES[gpu_model] * MACHINE_MAX_PRICE_RATE
    (default 2.0). Over the limit => actual_score=0.0.'
- name: uptime
  metric_kind: uptime
  direction: higher_is_better
  weight: null
  normalization: none
  notes: uptime_multiplier = 1 - PORTION_FOR_UPTIME + PORTION_FOR_UPTIME * min(1, uptime/UPTIME_REQUIRED_MINUTES);
    UPTIME_REQUIRED_MINUTES = 60*24*14 (14 days). Skipped when SKIP_COLLATERAL_PENALTY or collateral deposited.
ground_truth_sources:
- kind: hardware_probe
  description: Validator SSHes into the executor and runs device-info-seeded matrix-multiplication (GraVal/DMCompVerify),
    VerifyX network/storage/memory probes, and GPU spec/utilization reads. The probe results are the ground
    truth for hardware authenticity and capacity.
  source_identifier: matrix_validation_service.py / verifyx_validation_service.py
  trust_model: adversarial
- kind: market_data
  description: Rental status and price are sourced from the backend (rented executors API) and per-GPU
    reference prices in MACHINE_PRICES; collateral deposits are read from an on-chain contract.
  source_identifier: MACHINE_PRICES / collateral_contract_service.py
  trust_model: trusted
aggregation:
  method: proportional
  normalization: sum_to_one
  decay_rate: 0.3
  min_weight_floor: null
  burn_allocation:
    enabled: true
    address_or_uid: null
    fraction: 0.91
    notes: 'TOTAL_BURN_EMISSION = 0.91 routed to burner UIDs (settings.NEW_BURNERS, equal split when ENABLE_NEW_BURN_LOGIC).
      mining_share = 1 - 0.91 = 0.09 distributed across miners proportional to mining_score. address_or_uid
      null: burner UIDs are runtime config (settings.NEW_BURNERS).'
  notes: 'Per-executor: result.incentive = mining_share * mining_score / total_mining_score. Per-cycle
    incentives accumulate into miner_scores across cycles, then are normalized via process_weights_for_netuid
    at set_weights. decay_rate 0.3 is the VerifyX network-speed EMA alpha, not a score-level temporal
    decay.'
weight_setting:
  cadence: per_epoch
  method: Accumulated per-cycle miner_scores (mining incentives + burn shares) are normalized with bittensor
    process_weights_for_netuid, converted to uint weights, and committed via set_weights. Jobs run every
    BLOCKS_FOR_JOB blocks; weights are set when last_update >= tempo.
  on_chain_call: set_weights
  tempo_or_interval: tempo (TEMPO=360 blocks per epoch); set when last_update >= tempo
anti_gaming:
- kind: hardware_validation
  enforcement: rejection
  description: 'GraVal/DMCompVerify device-info-seeded matrix challenge proves GPU authenticity: the validator
    encrypts a challenge seeded with gpu uuids/count/model; the executor must decrypt and return the embedded
    UUID. UUID mismatch fails the job (score 0). Also LIB_NVIDIA_ML_DIGESTS attestation and TDX_WHITELIST
    hash checks.'
- kind: collateral
  enforcement: penalty
  description: 'Miners must deposit a per-GPU-type collateral (REQUIRED_DEPOSIT_AMOUNT, on-chain contract).
    If not deposited and collateral enforced: actual_score=job_score=0. With ENABLE_NO_COLLATERAL the
    deposit becomes a warning; IS_NOT_DEPOSITED_SCORE_MULTIPLIER=0.5 and an uptime ramp apply instead.
    Outdated contract version multiplies score by SCORE_PORTION_FOR_OLD_CONTRACT=0.'
- kind: deterministic_check
  enforcement: rejection
  description: 'GPU utilization gate: an unrented executor reporting gpu_utilization >= GPU_UTILIZATION_LIMIT
    (5%) or memory utilization > GPU_MEMORY_UTILIZATION_LIMIT (5%) is treated as already-busy/rented and
    rejected from the unrented scoring pool. MAX_GPU_COUNT=14 caps reported GPUs.'
- kind: rate_limit
  enforcement: penalty
  description: 'Price gate: price_per_gpu above MACHINE_PRICES[gpu_model]*MACHINE_MAX_PRICE_RATE (2.0)
    sets actual_score to 0.0, preventing miners from advertising overpriced rentals.'
sub_competitions:
  structure: multi_mechanism
  stages:
  - name: default_incentive
    role: main
    advancement_rule: 'DefaultIncentive: per-GPU mining_score with 0.91 fixed burn and 0.09 mining share.'
    multipliers: {}
  - name: rental_price_incentive
    role: main
    advancement_rule: Alternative IncentiveFactory algorithm rewarding actual rentals; uses dynamic burn_share
      = TOTAL_BURN_EMISSION - rental_share with per-GPU-type unrented caps (MAX_UNRENTED_GPUS_BY_TYPE).
    multipliers: {}
  count: 2
  notes: Incentive algorithm is selected at runtime via settings.incentive / IncentiveFactory.create.
    The active algorithm in default config is DefaultIncentive; rental_price is an alternative.
